22 int GenerateClassificationDataset::LoadSplitImages(const path& split)
23 {
24  int n_samples = 0;
25  int label_counter = -1;
26  path label;
27  path root_directory = dataset_root_directory_ / split;
28 
29  for (auto& p : recursive_directory_iterator(root_directory)) {
30  if (is_directory(p.path())) {
31  label = p.path().stem();
32  ++label_counter;
33  }
34  else {
35  if (p.path().parent_path().stem() == label) {
36  Sample s;
37  s.location_.emplace_back(root_directory.stem() / label / p.path().filename().generic_string());
38  s.label_ = vector<int>{ label_counter };
39  ++n_samples;
40  d_.samples_.push_back(s);
41  }
42  else {
43  cout << ECVL_WARNING_MSG "File '" + p.path().generic_string() + "' doesn't have an allowable label and it isn't added to the dataset file." << endl;
44  }
45  }
46  }

51 int GenerateSegmentationDataset::LoadSplitImages(const path& split)
52 {
53  int n_samples = 0;
54  bool gt_exist = false;
55  string images, ground_truth;
56  path root_directory = dataset_root_directory_ / split;
57 
58  if (exists(root_directory / "images")) {
59  images = "images";
60  ground_truth = "ground_truth";
61  }
62  else {
63  images = ground_truth = "";
64  }
65  path gt_img;
66 
67  for (auto& p : directory_iterator(root_directory / images)) {
68  gt_exist = false;
69  if (!is_directory(p.path()) && p.path().extension() != ".yml") {
70  string img = (split / images / p.path().filename()).generic_string();
71 
72  Sample s;
73  s.location_.emplace_back(img);
74 
75  if (suffix_.empty() && gt_name_.empty()) {
76  gt_img = root_directory / ground_truth / p.path().filename();
77  if (exists(gt_img) && !ground_truth.empty()) {
78  s.label_path_ = (split / ground_truth / gt_img.filename()).generic_string();
79  gt_exist = true;
80  }
81  else if (split != "test") { // test may not have ground truth images
82  if (splits_.empty()) {
83  if (!ground_truth.empty()) {
84  cout << ECVL_WARNING_MSG "File '" + gt_img.generic_string() + "' doesn't exist and it isn't added to the dataset file." << endl;
85  }
86  }
87  else {
88  cout << ECVL_WARNING_MSG "File '" + img + "' doesn't have an existing ground truth and it isn't added to the dataset file." << endl;
89  }
90  }
91  }
92  else {
93  if ((!suffix_.empty() && img.find(suffix_.string()) != string::npos) ||
94  (!gt_name_.empty() && img.find(gt_name_.string()) != string::npos)) {
95  // suffix found, it is a gt -> skip
96  continue;
97  }
98 
99  gt_img = root_directory / ground_truth / path(p.path().stem().string() + suffix_.string());
100 
101  if (exists(gt_img)) {
102  s.label_path_ = (split / ground_truth / gt_img.filename()).generic_string();
103  gt_exist = true;
104  }
105  else if (exists(root_directory / ground_truth / gt_name_) && !gt_name_.empty() ) {
106  s.label_path_ = (split / ground_truth / gt_name_).generic_string();
107  gt_exist = true;
108  }
109  else if (split != "test") {
110  if (splits_.empty()) {
111  cout << ECVL_WARNING_MSG "File '" + gt_img.generic_string() + "' doesn't exist and it isn't added to the dataset file." << endl;
112  }
113  else {
114  cout << ECVL_WARNING_MSG "File '" + img + "' doesn't have an existing ground truth and it isn't added to the dataset file." << endl;
115  }
116  }
117  }
118 
119  // for training and validation splits it is mandatory to have ground truth images
120  if (split == "test" || gt_exist || splits_.empty()) {
121  d_.samples_.push_back(s);
122  ++n_samples;
123  }
124  }
125  }
126 
127  return n_samples;
128 }

130 void GenerateDataset::LoadImagesAndSplits()
131 {
132  if (splits_.empty()) {
133  // load locations and labels
134  this->LoadSplitImages("");
135  }
136  else {

139  // load locations and labels and get the number of samples for each split

141  num_samples_[i] = this->LoadSplitImages(splits_[i]);
142  }
143 
144  // load indexes of images for each split
145  for (int i = 0; i < splits_.size(); ++i) {
146  if (splits_[i] == "training") {
147  d_.split_.training_.resize(num_samples_[i]);
148  iota(d_.split_.training_.begin(), d_.split_.training_.end(), img_index);
149  }
150  else if (splits_[i] == "validation") {
151  d_.split_.validation_.resize(num_samples_[i]);
152  iota(d_.split_.validation_.begin(), d_.split_.validation_.end(), img_index);
153  }
154  else if (splits_[i] == "test") {
155  d_.split_.test_.resize(num_samples_[i]);
156  iota(d_.split_.test_.begin(), d_.split_.test_.end(), img_index);
157  }
158  img_index += num_samples_[i];
159  }
160  }
