195 void GaussianBlur(const Image& src, Image& dst, int sizeX, int sizeY, double sigmaX, double sigmaY)
196 {
197  AlwaysCheck(src, dst);
198 
199  if (src.channels_ != "xyc") {
200  ECVL_ERROR_NOT_IMPLEMENTED
201  }
202 
203  if (sizeX < 0 || (sizeX % 2 != 1)) {
204  ECVL_ERROR_WRONG_PARAMS("sizeX must either be positive and odd or zero")
205  }
206  if (sizeY < 0 || (sizeY % 2 != 1)) {
207  ECVL_ERROR_WRONG_PARAMS("sizeY must either be positive and odd or zero")
208  }
209 
210  bool sigmaX_zero = false;
211  if (sigmaX <= 0) {
212  sigmaX_zero = true;
213  if (sizeX == 0) {
214  ECVL_ERROR_WRONG_PARAMS("sigmaX and sizeX can't be both 0")
215  }
216  else {
217  sigmaX = 0.3 * ((sizeX - 1) * 0.5 - 1) + 0.8;
218  }
219  }
220  if (sigmaY <= 0) {
221  if (!sigmaX_zero) {
222  sigmaY = sigmaX;
223  }
224  else if (sizeY == 0) {
225  ECVL_ERROR_WRONG_PARAMS("sigmaX, sigmaY and sizeY can't be 0 at the same time")
226  }
227  else {
228  sigmaY = 0.3 * ((sizeY - 1) * 0.5 - 1) + 0.8;
229  }
230  }
231 
232  src.hal_->GaussianBlur(src, dst, sizeX, sizeY, sigmaX, sigmaY);
233 }

235 void GaussianBlur(const Image& src, Image& dst, double sigma)
236 {
237  // Formula from: https://docs.opencv.org/3.1.0/d4/d86/group__imgproc__filter.html#gac05a120c1ae92a6060dd0db190a61afa
238  int size = static_cast<int>(((sigma - 0.8) / 0.3 + 1) / 0.5 + 1);
239 
240  // Check if computed size is even
241  size = size % 2 == 0 ? size + 1 : size;
242  // Check if computed size is less than 3, the smallest kernel size allowed
243  size = size < 3 ? 3 : size;
244 
245  GaussianBlur(src, dst, size, size, sigma, sigma);
246 }
