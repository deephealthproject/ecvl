201 class ProducersConsumerQueue
202 {
203  std::condition_variable cond_notempty_;
204  std::condition_variable cond_notfull_;
205  std::mutex mutex_;
206  std::queue<std::tuple<Sample, Image, Label*>> cpq_;
207  unsigned max_size_;
208  unsigned threshold_;
210 public:
211  ProducersConsumerQueue() {}
215  ProducersConsumerQueue(unsigned mxsz) : max_size_(mxsz), threshold_(max_size_ / 2) {}
220  ProducersConsumerQueue(unsigned mxsz, unsigned thresh) : max_size_(mxsz), threshold_(thresh) {}
221 
230  void Push(const Sample& sample, const Image& image, Label* const label)
231  {
232  std::unique_lock<std::mutex> lock(mutex_);
233  cond_notfull_.wait(lock, [this]() { return !IsFull(); });
234  cpq_.push(make_tuple(sample, image, label));
235  cond_notempty_.notify_one();
236  }
237 
247  void Pop(Sample& sample, Image& image, Label*& label)
248  {
249  std::unique_lock<std::mutex> lock(mutex_);
250  cond_notempty_.wait(lock, [this]() { return !IsEmpty(); });
251  std::tie(sample, image, label) = cpq_.front();
252  cpq_.pop();
253  if (Length() < threshold_) {
254  cond_notfull_.notify_one();
255  }
256  }
257 
262  bool IsFull() const
263  {
264  return cpq_.size() >= max_size_;
265  }
266 
271  bool IsEmpty() const
272  {
273  return cpq_.empty();
274  }
275 
280  size_t Length() const
281  {
282  return cpq_.size();
283  }
284 
290  void SetSize(int max_size, int thresh = -1)
291  {
292  max_size_ = max_size;
293  threshold_ = thresh != -1 ? thresh : max_size / 2;
294  }
295 
296  void Clear()
297  {
298  std::unique_lock<std::mutex> lock(mutex_);
299 
300  // Remove residual samples and delete data
301  while (!IsEmpty()) {
302  auto [sample, image, label] = cpq_.front();
303  delete label; // Deallocate pointer
304  cpq_.pop();
305  }
306  }

388  queue_{ static_cast<unsigned>(batch_size_ * queue_ratio_size * num_workers_) }
389  {
390  // resize current_batch_ to the number of splits and initialize it with 0
391  current_batch_.resize(split_.size(), 0);
392 
393  // Initialize n_channels_

395  n_channels_ = tmp.Channels();
396 
397  if (!split_.empty()) {

399  // Initialize resize_dims_ after that augmentations on the first image are performed
400  if (augs_.IsEmpty()) {
401  cout << ECVL_WARNING_MSG << "Augmentations are empty!" << endl;
402  }
403  else {
404  while (!augs_.Apply(current_split_, tmp)) {
405  ++current_split_;
406  }
407  }
408  auto y = tmp.channels_.find('y');
409  auto x = tmp.channels_.find('x');
410  assert(y != std::string::npos && x != std::string::npos);
411  resize_dims_.insert(resize_dims_.begin(), { tmp.dims_[y],tmp.dims_[x] });
412

415  n_channels_gt_ = samples_[0].LoadImage(ctype_gt_, true).Channels();

417  }

419  cout << ECVL_WARNING_MSG << "Missing splits in the dataset file." << endl;
420  }
421 
422  // Set drop_last parameter for each split
423  if (!drop_last.empty() && vsize(drop_last) == vsize(split_)) {
424  for (int i = 0; i < vsize(drop_last); ++i) {
425  split_[i].drop_last_ = drop_last[i];
426  }
427  }
428 
429  // Initialize num_batches, last_batch and the ThreadCounters for each split
430  auto s_index = 0;
431  splits_tc_ = std::vector<std::vector<ThreadCounters>>(vsize(split_));
432  for (auto& s : split_) {
433  s.SetNumBatches(batch_size_);
434  s.SetLastBatch(batch_size_);

436  InitTC(s_index);
437  ++s_index;
438  }
439 
440  switch (task_) {

442  tensors_shape_ = make_pair<vector<int>, vector<int>>(
443  { batch_size_, n_channels_, resize_dims_[0], resize_dims_[1] },
