204 class ProducersConsumerQueue
205 {
206  std::condition_variable cond_notempty_;
207  std::condition_variable cond_notfull_;
208  mutable std::mutex mutex_;
209  std::queue<std::tuple<Sample, Image, Label*>> cpq_;
210  unsigned max_size_;
211  unsigned threshold_;
213 public:
214  ProducersConsumerQueue() {}
218  ProducersConsumerQueue(unsigned mxsz) : max_size_(mxsz), threshold_(max_size_ / 2) {}
223  ProducersConsumerQueue(unsigned mxsz, unsigned thresh) : max_size_(mxsz), threshold_(thresh) {}
224 
233  void Push(const Sample& sample, const Image& image, Label* const label)
234  {
235  std::unique_lock<std::mutex> lock(mutex_);
236  cond_notfull_.wait(lock, [this]() { return cpq_.size() < max_size_; });
237  cpq_.push(make_tuple(sample, image, label));
238  cond_notempty_.notify_one();
239  }
240 
250  void Pop(Sample& sample, Image& image, Label*& label)
251  {
252  std::unique_lock<std::mutex> lock(mutex_);
253  cond_notempty_.wait(lock, [this]() { return !cpq_.empty(); });
254  std::tie(sample, image, label) = cpq_.front();
255  cpq_.pop();
256  if (cpq_.size() < threshold_) {
257  cond_notfull_.notify_all();
258  }
259  }
260 
265  bool IsFull() const
266  {
267  std::unique_lock<std::mutex> lock(mutex_);
268  return cpq_.size() >= max_size_;
269  }
270 
275  bool IsEmpty() const
276  {
277  std::unique_lock<std::mutex> lock(mutex_);
278  return cpq_.empty();
279  }
280 
285  size_t Length() const
286  {
287  std::unique_lock<std::mutex> lock(mutex_);
288  return cpq_.size();
289  }
290 
296  void SetSize(int max_size, int thresh = -1)
297  {
298  max_size_ = max_size;
299  threshold_ = thresh != -1 ? thresh : max_size / 2;
300  }
301 
302  void Clear()
303  {
304  std::unique_lock<std::mutex> lock(mutex_);
305 
306  // Remove residual samples and delete data
307  while (!cpq_.empty()) {
308  auto [sample, image, label] = cpq_.front();
309  delete label; // Deallocate pointer
310  cpq_.pop();
311  }
312  }

383  const ColorType ctype_gt = ColorType::GRAY,
384  const unsigned num_workers = 1,
385  const double queue_ratio_size = 1.,
386  const vector<bool>& drop_last = {},
387  bool verify = false) :

394  ctype_gt_{ ctype_gt },
395  queue_{ static_cast<unsigned>(batch_size * queue_ratio_size * std::min(num_workers, processor_count_)) }
396  {

398  current_batch_.resize(split_.size(), 0);
399 
400  // Initialize n_channels_
401  Image tmp = samples_[0].LoadImage(ctype);
402  n_channels_ = tmp.Channels();
403 
404  if (!split_.empty()) {
405  current_split_ = 0;
406  // Initialize resize_dims_ after that augmentations on the first image are performed
407  if (augs_.IsEmpty()) {
408  cout << ECVL_WARNING_MSG << "Augmentations are empty!" << endl;
409  }
410  else {
411  while (!augs_.Apply(current_split_, tmp)) {
412  ++current_split_;
413  }
414  }
415  auto y = tmp.channels_.find('y');
416  auto x = tmp.channels_.find('x');
417  assert(y != std::string::npos && x != std::string::npos);
418  resize_dims_.insert(resize_dims_.begin(), { tmp.dims_[y],tmp.dims_[x] });
419

421  if (samples_[0].label_path_ != nullopt) {
422  n_channels_gt_ = samples_[0].LoadImage(ctype_gt_, true).Channels();
423  }
424  }
425  else {
426  cout << ECVL_WARNING_MSG << "Missing splits in the dataset file." << endl;
427  }
428 
429  // Set drop_last parameter for each split
430  if (!drop_last.empty()) {
431  if (vsize(drop_last) == vsize(split_)) {
432  for (int i = 0; i < vsize(drop_last); ++i) {
433  split_[i].drop_last_ = drop_last[i];
434  }

442  auto s_index = 0;
443  splits_tc_ = std::vector<std::vector<ThreadCounters>>(vsize(split_));
444  for (auto& s : split_) {
