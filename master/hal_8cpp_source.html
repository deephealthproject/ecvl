30 HardwareAbstractionLayer* HardwareAbstractionLayer::Factory(Device dev, bool shallow)

62 void HardwareAbstractionLayer::Copy(const Image& src, Image& dst)
63 {
64  assert(src.dev_ == dst.dev_);
65  if (src.contiguous_) {
66  dst.data_ = dst.hal_->MemAllocateAndCopy(src.datasize_, src.data_);
67  }
68  else {
69  // When copying a non contiguous image, we make it contiguous (is this choice ok?)
70  dst.contiguous_ = true;
71  dst.hal_->Create(dst);
72 
73  int ndims = vsize(src.dims_);
74  int sncd; // smallest non-contiguous dimension
75  for (sncd = 0; sncd < ndims && src.strides_[sncd] == dst.strides_[sncd]; ++sncd);
76 
77  std::vector<int> pos(ndims, 0);
78  uint8_t* dst_data_ptr = dst.data_;
79  const uint8_t* src_data_ptr = src.data_;
80  while (true) {
81  dst.hal_->MemCopy(dst_data_ptr, src_data_ptr, dst.strides_[sncd]);
82  dst_data_ptr += dst.strides_[sncd];
83 
84  // Increment source pointer
85  int dim;
86  for (dim = sncd; dim < ndims; ++dim) {
87  ++pos[dim];
88  src_data_ptr += src.strides_[dim];
89  if (pos[dim] != src.dims_[dim])
90  break;
91  // Back to dimension starting position
92  pos[dim] = 0;
93  src_data_ptr -= src.dims_[dim] * src.strides_[dim];
94  }
95  if (dim == ndims)
96  break;
97  }
98  }
99 }
