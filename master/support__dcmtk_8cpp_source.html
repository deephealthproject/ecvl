32 bool DicomRead(const std::string& filename, Image& dst) {
33 
34  bool return_value = true;
35 
36  DJDecoderRegistration::registerCodecs();
37 
38  DicomImage* image = new DicomImage(filename.c_str());
39  if (image == NULL) {
40  return_value = false;
41  }
42  else {
43  if (image->getStatus() == EIS_Normal) {
44  const uint8_t* pixelData = reinterpret_cast<const uint8_t*> ((image->getOutputData(8 /* maybe it doesn't alyaws fit */, 0, 1)));
45  if (pixelData != NULL) {
46  /* do something useful with the pixel data */
47  int x = static_cast<int>(image->getWidth());
48  int y = static_cast<int>(image->getHeight());
49 
50  ColorType color_type = ColorType::RGB;
51  int planes = 3;
52  EP_Interpretation interpretation = image->getPhotometricInterpretation();
53  if (interpretation == EP_Interpretation::EPI_Monochrome1 || interpretation == EP_Interpretation::EPI_Monochrome2) {
54  color_type = ColorType::GRAY;
55  planes = 1;
56  }
57 
58  dst.Create({ x, y, planes }, ecvl::DataType::uint8, "xyc", color_type);
59  memcpy(dst.data_, pixelData, x * y * planes);
60  }
61  }
62  else {
63  std::cerr << "Error: cannot load DICOM image (" << DicomImage::getString(image->getStatus()) << ")" << std::endl;
64  return_value = false;
65  }
66  }
67  delete image;
68 
69  if (!return_value) {
70  dst = Image();
71  }
72 
73  DJDecoderRegistration::cleanup();
74 
75  return return_value;
76 }

82 bool DicomWrite(const std::string& filename, const Image& src) {
83 
84  char uid[100];
85  DcmFileFormat fileformat;
86  DcmDataset* dataset = fileformat.getDataset();
87 
88  DcmTag tag;
89 
90  dataset->putAndInsertString(DCM_SOPClassUID, UID_SecondaryCaptureImageStorage);
91  dataset->putAndInsertString(DCM_SOPInstanceUID, dcmGenerateUniqueIdentifier(uid, SITE_INSTANCE_UID_ROOT));
92  dataset->putAndInsertString(DCM_PatientName, "John^Doe");
93 
94  uint16_t samples_per_pixel, planar_configuration, rows, columns, pixel_representation;
95  std::string photometric_interpretation;
96  bool floating_point = false;
97 
98  if (src.elemtype_ == DataType::int8 || src.elemtype_ == DataType::int16 || src.elemtype_ == DataType::int32 || src.elemtype_ == DataType::int64) {
99  pixel_representation = 1;
100  }
101  else if (src.elemtype_ == DataType::uint8 || src.elemtype_ == DataType::uint16 /*|| src.elemtype_ == DataType::uint32 || src.elemtype_ == DataType::uint64*/) {
102  pixel_representation = 0;
103  }
104  else if (src.elemtype_ == DataType::float32 || src.elemtype_ == DataType::float64) {
105  if (src.colortype_ != ColorType::GRAY) {
106  ECVL_ERROR_NOT_IMPLEMENTED
107  }
108  floating_point = true;
109  }
110 
111  if (src.colortype_ == ColorType::GRAY) {
112  samples_per_pixel = 1;
113  photometric_interpretation = "MONOCHROME2"; // MONOCHROME1: 0 = white MONOCHROME2: 0 = black
114  }
115  else if (src.colortype_ == ColorType::RGB /*|| src.colortype_ == ColorType::BGR*/) {
116  samples_per_pixel = 3;
117  photometric_interpretation = "RGB";
118  }
119  else {
120  ECVL_ERROR_NOT_IMPLEMENTED
121  }
122 
123  if (src.channels_ == "cxy") {
124  rows = src.dims_[2];
125  columns = src.dims_[1];
126  planar_configuration = 0;
127  }
128  else if (src.channels_ == "xyc") {
129  rows = src.dims_[1];
130  columns = src.dims_[0];
131  planar_configuration = 1;
132  }
133  else {
134  ECVL_ERROR_NOT_IMPLEMENTED
135  }
136 
137  // Prova per immagini RGB
138  dataset->putAndInsertUint16(DCM_SamplesPerPixel, samples_per_pixel);
139  dataset->putAndInsertString(DCM_PhotometricInterpretation, photometric_interpretation.c_str(), static_cast<uint32_t>(photometric_interpretation.length()));
140  if (samples_per_pixel > 1) {
141  dataset->putAndInsertUint16(DCM_PlanarConfiguration, planar_configuration);
142  }
143  dataset->putAndInsertUint16(DCM_Rows, rows);
144  dataset->putAndInsertUint16(DCM_Columns, columns);
145  dataset->putAndInsertUint16(DCM_BitsAllocated, src.elemsize_ * 8);
146  if (!floating_point) {
147  dataset->putAndInsertUint16(DCM_BitsStored, src.elemsize_ * 8);
148  dataset->putAndInsertUint16(DCM_HighBit, src.elemsize_ * 8 - 1);
149  dataset->putAndInsertUint16(DCM_PixelRepresentation, pixel_representation);
150  }
151  dataset->putAndInsertString(DCM_LossyImageCompression, "00");
152 
153  if (!floating_point) {
154  dataset->putAndInsertUint8Array(DCM_PixelData, src.data_, src.datasize_);
155  }
156  else if (src.elemtype_ == DataType::float32) {
157  dataset->putAndInsertFloat32Array(DCM_FloatPixelData, reinterpret_cast<const float *>(src.data_), src.datasize_ / sizeof(float));
158  }
159  else if (src.elemtype_ == DataType::float64) {
160  dataset->putAndInsertFloat64Array(DCM_DoubleFloatPixelData, reinterpret_cast<const double*>(src.data_), src.datasize_ / sizeof(double));
161  }
162 
163  OFCondition status = fileformat.saveFile(filename.c_str(), EXS_LittleEndianExplicit);
164  if (status.bad())
165  return false;
166 
167  return true;
168 
169 }
