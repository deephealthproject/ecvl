169 template <class _RunIt>
170 static vector<uint8_t> CompressOverlay(_RunIt first, _RunIt last, size_t size) {
171 
172  vector<uint8_t> out((size + 7) / 8, uint8_t(0));
173  int i = 0;
174  int pos = 0;
175  for (auto it = first; it != last; ++it) {
176  uint8_t x = *it;
177  if (x != 0) {
178  out[i] |= 1u << pos;
179  }
180  pos++;
181  if (pos == 8) {
182  pos = 0;
183  i++;
184  }
185  }
186  return out;
187 }
