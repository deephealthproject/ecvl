599 
600  View(Image& img, const std::vector<int>& start, const std::vector<int>& size) : View(img)
601  {
602  dims_.clear();
603  int ssize = vsize(size);
604  for (int i = 0; i < ssize; ++i) {
605  if (start[i] < 0 || start[i] >= img.dims_[i])
606  throw std::runtime_error("Start of crop outside image limits");
607  dims_.push_back(img.dims_[i] - start[i]);
608  if (size[i] > dims_[i]) {
609  throw std::runtime_error("Crop outside image limits");
610  }
611  if (size[i] >= 0) {
612  dims_[i] = size[i];
613  }
614  }
615 
616  // Check if image has a color dimension
617  auto cpos = channels_.find('c');
618  if (cpos != std::string::npos) {
619  // If we are cropping the color channel, we fix the color information
620  if (dims_[cpos] != img.dims_[cpos]) {
621  if (dims_[cpos] == 1) {
622  colortype_ = ColorType::GRAY;
623  }
624  else {
625  channels_[cpos] = 'o';
626  colortype_ = ColorType::none;
627  }
628  }
629  }
630 
631  data_ = img.Ptr(start);
632  datasize_ = 0; // This is set to zero, because when the View is not contiguous, it's useless to relay on this information
633  contiguous_ = false;
634  }

636  basetype& operator()(const std::vector<int>& coords)
637  {
638  return *reinterpret_cast<basetype*>(Ptr(coords));
639  }
