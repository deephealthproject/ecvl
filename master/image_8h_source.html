514  View(Image& img, const std::vector<int>& start, const std::vector<int>& size) : View(img)
515  {
516  dims_.clear();
517  int ssize = vsize(size);
518  for (int i = 0; i < ssize; ++i) {
519  if (start[i] < 0 || start[i] >= img.dims_[i])
520  throw std::runtime_error("Start of crop outside image limits");
521  dims_.push_back(img.dims_[i] - start[i]);
522  if (size[i] > dims_[i]) {
523  throw std::runtime_error("Crop outside image limits");
524  }
525  if (size[i] >= 0) {
526  dims_[i] = size[i];
527  }
528  }
529 
530  // Check if image has a color dimension
531  auto cpos = channels_.find('c');
532  if (cpos != std::string::npos) {
533  // If we are cropping the color channel, we fix the color information
534  if (dims_[cpos] != img.dims_[cpos]) {
535  if (dims_[cpos] == 1) {
536  colortype_ = ColorType::GRAY;
537  }
538  else {
539  channels_[cpos] = 'o';
540  colortype_ = ColorType::none;
541  }
542  }
543  }
544 
545  data_ = img.Ptr(start);
546  datasize_ = 0; // This is set to zero, because when the View is not contiguous, it's useless to relay on this information
547  contiguous_ = false;
548  }

555  void Create(const std::vector<int>& dims, std::string channels, ColorType colortype, uint8_t* ptr, const std::vector<float>& spacings = std::vector<float>())
556  {
557  // Compute datasize
558  size_t new_datasize = DataTypeSize(DT);
559  new_datasize = std::accumulate(begin(dims), end(dims), new_datasize, std::multiplies<size_t>());
560 
561  if (datasize_ != new_datasize) {
562  datasize_ = new_datasize;
563  }
564 
565  elemtype_ = DT;
566  elemsize_ = DataTypeSize(elemtype_);
567  dims_ = dims; // A check could be added to save this copy
568  spacings_ = spacings;
569  channels_ = std::move(channels);
570  colortype_ = colortype;
571  datasize_ = new_datasize;
572 
573  // Compute strides
574  strides_ = { elemsize_ };
575  int dsize = vsize(dims_);
576  for (int i = 0; i < dsize - 1; ++i) {
577  strides_.push_back(strides_[i] * dims_[i]);
578  }
579 
580  data_ = ptr;
581  mem_ = ShallowMemoryManager::GetInstance();
582  return;
