20 TEST(Imgproc, Thresholding)
21 {
22  Image x({ 2, 2, 1 }, DataType::uint8, "xyc", ColorType::GRAY);
23  View<DataType::uint8> x_v(x);
24  Image o;
25 
26  x_v({ 0,0,0 }) = 250; x_v({ 1,0,0 }) = 16;
27  x_v({ 0,1,0 }) = 15; x_v({ 1,1,0 }) = 200;

31

34

36  o_v = o;
