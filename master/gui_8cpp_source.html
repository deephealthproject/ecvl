152 wxImage WxFromImg(Image& img)
153 {
154  if (img.colortype_ != ColorType::RGB)
155  ChangeColorSpace(img, img, ColorType::RGB);
156 
157  if (img.channels_ != "cxy")
158  RearrangeChannels(img, img, "cxy");
159 
160  wxImage wx(img.dims_[1], img.dims_[2], (uint8_t*)malloc(img.datasize_), false);
161  uint8_t* wx_data = wx.GetData();
162 
163  if(img.contiguous_)
164  memcpy(wx_data, img.data_, img.datasize_);
165  else
166  {
167  auto i = img.Begin<uint8_t>(), e = img.End<uint8_t>();
168  for (; i != e; ++i) {
169  *wx_data = *i;
170  ++wx_data;
171  }
172  }
173 
174  return wx;
175 }
