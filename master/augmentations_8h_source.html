35 class param
36 {
37  static std::istream& read_until(std::istream& is, std::string& s, const std::string& list)
38  {
39  s.clear();
40  while (is.peek() && is && list.find(is.peek()) == list.npos) {
41  s += is.get();
42  }
43  return is;
44  }
45 
46  void read_vals(std::istream& is, char closing_char)
47  {
48  double val;
49  char next_char;
50  do {
51  is >> val;
52  if (!is) {
53  break;
54  }
55  vals_.push_back(val);
56  is >> next_char;
57  } while (next_char == ',');
58  if (!is || next_char != closing_char) {
59  std::cerr << "Error while reading values of parameter " << name_ << "\n"; // TODO: standardize
60  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
61  }
62  }
63 
64 public:
65  enum class type { range, vector, number, string };
66 
67  static const char* to_string(type t)
68  {
69  switch (t) {
70  case type::range: return "range";
71  case type::vector: return "vector";
72  case type::number: return "number";
73  case type::string: return "string";
74  default:
75  ECVL_ERROR_NOT_REACHABLE_CODE
76  }
77  }
78 
79  std::string name_;
80  type type_;
81  std::vector<double> vals_;
82  std::string str_;
83 
84  param() {}
85  param(std::istream& is)
86  {
87  is >> std::ws;
88  read_until(is, name_, " =");
89  char next_char;
90  is >> std::ws >> next_char;
91  if (next_char != '=') {
92  throw std::runtime_error("Cannot read parameter name"); // TODO: standardize
93  }
94  is >> std::ws;
95  next_char = is.peek();
96  if (next_char == '[') { // range
97  is.ignore();
98  type_ = type::range;
99  read_vals(is, ']');
100  }
101  else if (next_char == '(') { // vector
102  is.ignore();
103  type_ = type::vector;
104  read_vals(is, ')');
105  }
106  else if (next_char == '"') { // string
107  is.ignore();
108  type_ = type::string;
109  std::getline(is, str_, '"');
110  }
111  else {
112  type_ = type::number;
113  vals_.resize(1);
114  is >> vals_[0];
115  }
116  if (!is) {
117  std::cerr << "Error while reading value of parameter " << name_ << "\n"; // TODO: standardize
118  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
119  }
120  }
121 
122  friend class param_list;
123  static param_list read(std::istream& is, std::string fn_name_);
124 };

130 public:

138  bool Get(const std::string& name, param::type type, bool required, param& value)
139  {
140  auto it = m_.find(name);
141  if (it != end(m_)) {
142  auto& p = it->second;
143  if (p.type_ != type) {
144  throw std::runtime_error(fn_name_ + ": " + name + " parameter must be a " + param::to_string(type));
145  }
146  value = p;
147  return true;
148  }
149  if (required) {
150  throw std::runtime_error(fn_name_ + ": " + name + " is a required parameter");
151  }
152  return false;
