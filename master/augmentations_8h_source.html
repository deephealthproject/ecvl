34 class param
35 {
36  static std::istream& read_until(std::istream& is, std::string& s, const std::string& list)
37  {
38  s.clear();
39  while (is.peek() && is && list.find(is.peek()) == list.npos) {
40  s += is.get();
41  }
42  return is;
43  }
44 
45  void read_vals(std::istream& is, char closing_char)
46  {
47  double val;
48  char next_char;
49  do {
50  is >> val;
51  if (!is) {
52  break;
53  }
54  vals_.push_back(val);
55  is >> next_char;
56  } while (next_char == ',');
57  if (!is || next_char != closing_char) {
58  std::cerr << "Error while reading values of parameter " << name_ << "\n"; // TODO: standardize
59  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
60  }
61  }
62 
63 public:
64  enum class type { range, vector, number, string };
65 
66  static const char* to_string(type t)
67  {
68  switch (t) {
69  case type::range: return "range";
70  case type::vector: return "vector";
71  case type::number: return "number";
72  case type::string: return "string";
73  default:
74  ECVL_ERROR_NOT_REACHABLE_CODE
75  }
76  }
77 
78  std::string name_;
79  type type_;
80  std::vector<double> vals_;
81  std::string str_;
82 
83  param() {}
84  param(std::istream& is)
85  {
86  is >> std::ws;
87  read_until(is, name_, " =");
88  char next_char;
89  is >> std::ws >> next_char;
90  if (next_char != '=') {
91  throw std::runtime_error("Cannot read parameter name"); // TODO: standardize
92  }
93  is >> std::ws;
94  next_char = is.peek();
95  if (next_char == '[') { // range
96  is.ignore();
97  type_ = type::range;
98  read_vals(is, ']');
99  }
100  else if (next_char == '(') { // vector
101  is.ignore();
102  type_ = type::vector;
103  read_vals(is, ')');
104  }
105  else if (next_char == '"') { // string
106  is.ignore();
107  type_ = type::string;
108  std::getline(is, str_, '"');
109  }
110  else {
111  type_ = type::number;
112  vals_.resize(1);
113  is >> vals_[0];
114  }
115  if (!is) {
116  std::cerr << "Error while reading value of parameter " << name_ << "\n"; // TODO: standardize
117  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
118  }
119  }
120 
121  friend class param_list;
122  static param_list read(std::istream& is, std::string fn_name_);
123 };

128  const std::string fn_name_;
129 public:
130  param_list(std::string fn_name) : fn_name_(move(fn_name)) {}

133  {
134  return m_[s];

136 
137  bool Get(const std::string& name, param::type type, bool required, param& value)
138  {
139  auto it = m_.find(name);
140  if (it != end(m_)) {
141  auto& p = it->second;
142  if (p.type_ != type) {
143  throw std::runtime_error(fn_name_ + ": " + name + " parameter must be a " + param::to_string(type));
144  }
145  value = p;
146  return true;
147  }
148  if (required) {
149  throw std::runtime_error(fn_name_ + ": " + name + " is a required parameter");
150  }
151  return false;
152  }
