48 bool OpenSlideRead(const path& filename, Image& dst, const int level, const vector<int>& dims)
49 {
50  const int& x = dims[0];
51  const int& y = dims[1];
52  const int& w = dims[2];
53  const int& h = dims[3];
54 
55  bool open_status = true;
56 
57  openslide_t* osr = openslide_open(filename.string().c_str());
58 
59  if (osr == NULL || openslide_get_error(osr) != NULL) {
60  return false;
61  }
62  vector<uint32_t> d(sizeof(uint32_t) * w * h);
63  openslide_read_region(osr, d.data(), x, y, level, w, h);
64 
65  dst.Create({ 3, w, h }, DataType::uint8, "cxy", ColorType::BGR);
66 
67  uint8_t a, r, g, b;
68  uint32_t pixel;
69  for (int i = 0, j = 0; i < dst.datasize_; ++j, ++i) {
70  pixel = d[j];
71  a = pixel >> 24;
72  switch (a) {
73  case 0:
74  r = g = b = 0;
75  break;
76  case 255:
77  r = (pixel >> 16) & 0xff;
78  g = (pixel >> 8) & 0xff;
79  b = pixel & 0xff;
80  break;
81  default:
82  r = 255 * ((pixel >> 16) & 0xff) / a;
83  g = 255 * ((pixel >> 8) & 0xff) / a;
84  b = 255 * (pixel & 0xff) / a;
85  }
86  dst.data_[i] = b;
87  dst.data_[++i] = g;
88  dst.data_[++i] = r;
89  }
90  openslide_close(osr);
91  return !dst.IsEmpty();
92 }
