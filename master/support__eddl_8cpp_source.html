60 void ImageToTensor(const Image& img, tensor& t)
61 {
62  Image tmp;
63  string channels;
64 
65  if (img.dims_.size() != 3) {
66  cerr << ECVL_ERROR_MSG "Image must have 3 dimensions 'xy[czo]' (in any order)" << endl;
67  ECVL_ERROR_NOT_IMPLEMENTED
68  }
69 
70  // If img is one of: cxy, cyx, xcy, ycx... convert it to xyc
71  if (img.channels_.find('c') != string::npos && img.channels_ != "xyc") {
72  channels = "xyc";
73  }
74  // If img is one of: zxy, zyx, xzy, yzx... convert it to xyz
75  else if (img.channels_.find('z') != string::npos && img.channels_ != "xyz") {
76  channels = "xyz";
77  }
78  // If img is one of: oxy, oyx, xoy, yox... convert it to xyo
79  else if (img.channels_.find('o') != string::npos && img.channels_ != "xyo") {
80  channels = "xyo";
81  }
82  else if (img.channels_.find('o') == string::npos &&
83  img.channels_.find('c') == string::npos &&
84  img.channels_.find('z') == string::npos) {
85  ECVL_ERROR_NOT_IMPLEMENTED
86  }
87 
88  if (channels.size() > 0) {
89  RearrangeChannels(img, tmp, channels, DataType::float32);
90  }
91  else {
92  CopyImage(img, tmp, DataType::float32);
93  }
94  t = new Tensor({ tmp.dims_[2], tmp.dims_[1], tmp.dims_[0] });
95 
96  memcpy(t->ptr, tmp.data_, tmp.datasize_);
97 }

184 void DLDataset::LoadBatch(tensor& images, tensor& labels)
185 {
186  if (resize_dims_.size() != 2) {
187  cerr << ECVL_ERROR_MSG "resize_dims_ must have 2 dimensions (height, width)" << endl;
188  ECVL_ERROR_INCOMPATIBLE_DIMENSIONS
189  }
190 
191  int& bs = batch_size_;
192  Image img, gt;
193  int offset = 0, start = 0;
194 
195  // Check if tensors size matches with batch dimensions
196  // size of images tensor must be equal to batch_size * number_of_image_channels * image_width * image_height
197  if (images->size != bs * n_channels_ * resize_dims_[0] * resize_dims_[1]) {
198  cerr << ECVL_ERROR_MSG "images tensor must have N = batch_size, C = number_of_image_channels, H = image_height, W = image_width" << endl;
199  ECVL_ERROR_INCOMPATIBLE_DIMENSIONS
200  }
201 
202  // if it is a classification problem, size of labels tensor must be equal to batch_size * number_of_classes
203  if (samples_[0].label_.has_value()) {
204  if (labels->size != bs * classes_.size()) {
205  cerr << ECVL_ERROR_MSG "labels tensor must have N = batch_size, C = number_of_classes" << endl;
206  ECVL_ERROR_INCOMPATIBLE_DIMENSIONS
207  }
208  }
209  // otherwise is a segmentation problem so size of labels tensor must be equal to batch_size * number_of_label_channels * image_width * image_height
210  else if (labels->size != bs * n_channels_gt_ * resize_dims_[0] * resize_dims_[1]) {
211  cerr << ECVL_ERROR_MSG "labels tensor must have N = batch_size, C = number_of_label_channels, H = image_height, W = image_width" << endl;
212  ECVL_ERROR_INCOMPATIBLE_DIMENSIONS
213  }
214 
215  // Move to next samples
216  start = current_batch_[+current_split_] * bs;
217  ++current_batch_[+current_split_];
218 
219  if (vsize(GetSplit()) < start + bs) {
220  cerr << ECVL_ERROR_MSG "Batch size is not even with the number of samples. Hint: loop through `num_batches = num_samples / batch_size;`" << endl;
221  ECVL_ERROR_CANNOT_LOAD_IMAGE
222  }
223 
224  // Fill tensors with data
225  for (int i = start; i < start + bs; ++i) {
226  // Read the image
227  const int index = GetSplit()[i];
228  Sample& elem = samples_[index];
229  img = elem.LoadImage(ctype_, false);
230 
231  // Classification problem
232  if (elem.label_) {
233  // Apply chain of augmentations only to sample image
234  augs_.Apply(current_split_, img);
235 
236  // Copy label into tensor (labels)
237  vector<float> lab(classes_.size(), 0);
238  for (int j = 0; j < vsize(elem.label_.value()); ++j) {
239  lab[elem.label_.value()[j]] = 1;
240  }
241  memcpy(labels->ptr + lab.size() * offset, lab.data(), lab.size() * sizeof(float));
242  }
243  // Segmentation problem
244  else if (elem.label_path_) {
245  // Read the ground truth
246  gt = elem.LoadImage(ctype_gt_, true);
247 
248  // Apply chain of augmentations to sample image and corresponding ground truth
249  augs_.Apply(current_split_, img, gt);
250 
251  // Copy label into tensor (labels)
252  ImageToTensor(gt, labels, offset);
253  }
254 
255  // Copy image into tensor (images)
256  ImageToTensor(img, images, offset);
257 
258  ++offset;
259  }
260 }
