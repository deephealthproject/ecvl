43  default:
44  ECVL_ERROR_UNSUPPORTED_OPENCV_DEPTH

48  std::string channels;
49  if (m.dims < 2) {
50  ECVL_ERROR_UNSUPPORTED_OPENCV_DIMS
51  }
52  else if (m.dims == 2) {
53  channels = "xy";
54  }
55  else if (m.dims == 3) {
56  channels = "xyz";
57  }
58  else {
59  ECVL_ERROR_UNSUPPORTED_OPENCV_DIMS
60  }
61

69  channels += "c";
70  dims.push_back(3); // Add another dim for color planes
71  colortype = ColorType::BGR;
72  }
73  else if (m.channels() == 1) {
74  colortype = ColorType::none;
75  }

78  dims.push_back(m.channels()); // Add another dim for color planes
79  colortype = ColorType::none;
80  }

103 cv::Mat ImageToMat(const Image& img)
104 {
105  if (img.channels_ != "cxy" && img.channels_ != "xyc" &&
106  img.channels_ != "zxy" && img.channels_ != "xyz" &&
107  img.channels_ != "oxy" && img.channels_ != "xyo") {
108  ECVL_ERROR_NOT_IMPLEMENTED
109  }
110  if (img.colortype_ != ColorType::BGR && img.colortype_ != ColorType::GRAY) {
111  ECVL_ERROR_NOT_IMPLEMENTED
112  }
113 
114  Image tmp;
115  if (img.channels_.find('c') != std::string::npos) {
116  RearrangeChannels(img, tmp, "cxy");
117  }
118  else if (img.channels_.find('z') != std::string::npos) {
119  RearrangeChannels(img, tmp, "zxy");
120  }
121  else if (img.channels_.find('o') != std::string::npos) {
122  RearrangeChannels(img, tmp, "oxy");
123  }
124 
125  int type;
126  switch (tmp.elemtype_) {
127  case DataType::uint8: type = CV_MAKETYPE(CV_8U, tmp.dims_[0]); break;
128  case DataType::int8: type = CV_MAKETYPE(CV_8S, tmp.dims_[0]); break;
129  case DataType::uint16: type = CV_MAKETYPE(CV_16U, tmp.dims_[0]); break;
130  case DataType::int16: type = CV_MAKETYPE(CV_16S, tmp.dims_[0]); break;
131  case DataType::int32: type = CV_MAKETYPE(CV_32S, tmp.dims_[0]); break;
132  case DataType::float32: type = CV_MAKETYPE(CV_32F, tmp.dims_[0]); break;
133  case DataType::float64: type = CV_MAKETYPE(CV_64F, tmp.dims_[0]); break;
134  default:
135  break;
136  }
137 
138  int mdims[] = { tmp.dims_[2], tmp.dims_[1] }; // Swap dimensions to have rows, cols
139 
140  cv::Mat m(2, mdims, type);
141  memcpy(m.data, tmp.data_, tmp.datasize_);
142 
143  return m;
144 }
