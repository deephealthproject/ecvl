67  void SwitchEndianness(char* data, size_t len)
68  {
69  for (size_t i = 0; i < len / 2; i++) {
70  std::swap(data[i], data[len - i - 1]);
71  }
72  }
73 
74  void ReadSwapEndianness(char* dst, size_t len = 1, long long n = 1)
75  {
76  is_.read(dst, len * n);
77  if (len > 1) {
78  for (int i = 0; i < n; i++) {
79  SwitchEndianness(dst + i * len, len);
80  }
81  }
82  }
83 
84  void operator()(char* dst, size_t len = 1, long long n = 1)
85  {
86  if (swap_endianness_) {
87  ReadSwapEndianness(dst, len, n);

90  is_.read(dst, len * n);
91  }

443  }

447  case DataType::uint16: datatype = DT_UINT16; break;
