23 float SincNorm(float x)
24 {
25  if (fabs(x) < 1e-7) {
26  return 1.0f - fabs(x);
27  }
28  double y = M_PI * x;
29  return static_cast<float>(sin(y) / y);

32 float Hanning(float x, int half_w)
33 {
34  if (fabs(x) > half_w) {
35  return 0;
36  }
37  else {
38  return static_cast<float>(0.5 + 0.5 * cos(M_PI * x / half_w));
39  }
40 }

42 Image SincKernel(const string& sincwindowtype, int w, int n)
43 {
44  int nstore = n;
45  if (nstore < 1) {
46  nstore = 1;
47  }
48  Image ker({ nstore, 1, 1 }, DataType::float32, "xyc", ColorType::GRAY);
49  auto it = ker.ContiguousBegin<float>(), end = ker.ContiguousEnd<float>();
50 
51  int half_w = (w - 1) / 2;
52  float half_nstore = (nstore - 1) / 2.0f;
53 
54  for (int n = 0; n < nstore; ++n, ++it) {
55  float x = (n - half_nstore) / half_nstore * half_w;
56  if ((sincwindowtype == "hanning") || (sincwindowtype == "h")) {
57  *it = SincNorm(x) * Hanning(x, half_w);
58  }
59  else {
60  cerr << "ERROR: Unrecognised sinc window type - using hanning" << endl;
61  ker = SincKernel("h", w, nstore);
62  return ker;
63  }
64  }
65  return ker;
66 }

68 float KernelVal(float x, int w, const Image& kernel)
69 {
70  // linearly interpolates to get the kernel at the point (x)
71  // given the half-width w
72  if (fabs(x) > w) return 0.0;
73  float halfnk = (kernel.Width() - 1) / 2.0f;
74  float dn = x / w * halfnk + halfnk + 1.0f;
75  int n = (int)floor(dn);
76  dn -= n;
77  if (n > (kernel.Width() - 1)) return 0.0;
78  if (n < 1) return 0.0;
79 
80  float* kernel_data = reinterpret_cast<float*>(kernel.data_);
81 
82  return kernel_data[n - 1] * (1.0f - dn) + kernel_data[n] * dn;
83 }

90 float Extrapolate(const Image& data, const int index)
91 {
92  float extrapval;
93  float* data_data = reinterpret_cast<float*>(data.data_);
94  int n_volumes = data.dims_[3];
95 
96  if (InBounds(n_volumes, index))
97  extrapval = data_data[index - 1];
98  else if (InBounds(n_volumes, index - 1))
99  extrapval = data_data[data.Width() - 1];
100  else if (InBounds(n_volumes, index + 1))
101  extrapval = data_data[0];
102  else {
103  auto it = data.Begin<float>();
104  auto e = data.End<float>();
105  double sum = 0;
106  for (; it != e; ++it) {
107  sum += *it;
108  }
109  extrapval = static_cast<float>(sum / std::accumulate(std::begin(data.dims_), std::end(data.dims_), 1, std::multiplies<int>()));
110  }
111 
112  return extrapval;
113 }

115 float KernelInterpolation(const Image& data, float index, const Image& userkernel, int width)
116 {
117  int widthx = (width - 1) / 2;
118  // kernel half-width (i.e. range is +/- w)
119  int ix0;
120  ix0 = (int)floor(index);
121  int n_volumes = data.dims_[3];
122 
123  int wx(widthx);
124  vector<float> storex(2 * wx + 1);
125  for (int d = -wx; d <= wx; d++)
126  storex[d + wx] = KernelVal((index - ix0 + d), wx, userkernel);
127 
128  float convsum = 0.0, interpval = 0.0, kersum = 0.0;
129 
130  int xj;
131  float* data_data = reinterpret_cast<float*>(data.data_);
132 
133  for (int x1 = ix0 - wx; x1 <= ix0 + wx; x1++) {
134  if (InBounds(n_volumes, x1)) {
135  xj = ix0 - x1 + wx;
136  float kerfac = storex[xj];
137  convsum += data_data[x1 - 1] * kerfac;
138  kersum += kerfac;
139  }
140  }
141 
142  if ((fabs(kersum) > 1e-9)) {
143  interpval = convsum / kersum;
144  }
145  else {
146  interpval = (float)Extrapolate(data, ix0);
147  }
148  return interpval;
149 }

151 void CpuHal::SliceTimingCorrection(const Image& src, Image& dst, bool odd, bool down)
152 {
153  Image timeseries;
154  ecvl::CopyImage(src, timeseries, DataType::float32);
155 
156  int no_volumes, no_slices;
157  float repeat_time, offset = 0, slice_spacing;
158 
159  auto t = timeseries.channels_.find('t');
160  no_volumes = timeseries.dims_[t];
161  repeat_time = timeseries.spacings_[t];
162 
163  if (repeat_time == 0) {
164  repeat_time = 3;
165  }
166 
167  auto z = timeseries.channels_.find('z');
168  no_slices = timeseries.dims_[z];
169  slice_spacing = repeat_time / no_slices;
170 
171  Image userkernel = SincKernel("hanning", 7, 1201);
172 
173  float recenter = (((float)no_slices) / 2 - 0.5f) / no_slices;
174 
175  for (int slice = 1; slice <= no_slices; slice++) {
176  if (odd) {
177  if ((slice % 2) == 0)
178  offset = recenter - (ceil((float)no_slices / 2) + ((slice - 1) / 2)) * (slice_spacing / repeat_time);
179  else
180  offset = recenter - ((slice - 1) / 2) * (slice_spacing / repeat_time);
181  }
182  else if (down) {
183  offset = recenter - (no_slices - slice) * (slice_spacing / repeat_time);
184  }
185  else {
186  offset = recenter - (slice - 1) * (slice_spacing / repeat_time);
187  }
188 
189  for (int x_pos = 0; x_pos < timeseries.Width(); x_pos++)
190  for (int y_pos = 0; y_pos < timeseries.Height(); y_pos++) {
191  View<DataType::float32> tmp_v(timeseries, { 0,0,slice - 1,0 }, { -1,-1,-1,1 });
192  View<DataType::float32> voxeltimeseries(timeseries, { x_pos, y_pos, slice - 1, 0 }, { 1, 1, 1, -1 });
193  Image voxeltimeseries_image = voxeltimeseries;
194  Image interpseries = voxeltimeseries;
195  auto it_int = interpseries.ContiguousBegin<float>();
196  for (int time_step = 1; time_step <= no_volumes; time_step++, ++it_int) {
197  *it_int = KernelInterpolation(voxeltimeseries_image, time_step + offset, userkernel, 7);
198  }
199  float* inter_data = reinterpret_cast<float*>(interpseries.data_);
200 
201  for (int t = 0; t < no_volumes; t++) {
202  *reinterpret_cast<float*>(timeseries.Ptr({ x_pos, y_pos, slice - 1, t })) = inter_data[t];
203  }
204  }
205  }
206 
207  dst = std::move(timeseries);
208 }
