37 template <DataType DT> // src type
38 struct NormalizeToUint8Struct
39 {
40  static void _(const Image& src, Image& dst)
41  {
42  dst.Create(src.dims_, DataType::uint8, src.channels_, src.colortype_, src.spacings_);
43 
44  ConstView<DT> src_v(src);
45  View<DataType::uint8> dst_v(dst);
46 
47  // find max and min
48  TypeInfo_t<DT> max = *std::max_element(src_v.Begin(), src_v.End());
49  TypeInfo_t<DT> min = *std::min_element(src_v.Begin(), src_v.End());
50 
51  auto dst_it = dst_v.Begin();
52  auto src_it = src_v.Begin();
53  auto src_end = src_v.End();
54  for (; src_it != src_end; ++src_it, ++dst_it) {
55  (*dst_it) = (((*src_it) - min) * 255) / (max - min);
56  }
57  }
58 };
