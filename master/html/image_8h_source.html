508  View(Image& img, const std::vector<int>& start, const std::vector<int>& size) : View(img)
509  {
510  dims_.clear();
511  int ssize = size.size();
512  for (int i = 0; i < ssize; ++i) {
513  if (start[i] < 0 || start[i] >= img.dims_[i])
514  throw std::runtime_error("Start of crop outside image limits");
515  dims_.push_back(img.dims_[i] - start[i]);
516  if (size[i] > dims_[i]) {
517  throw std::runtime_error("Crop outside image limits");
518  }
519  if (size[i] >= 0) {
520  dims_[i] = size[i];
521  }
522  }
523 
524  // Check if image has a color dimension
525  auto cpos = channels_.find('c');
526  if (cpos != std::string::npos) {
527  // If we are cropping the color channel, we fix the color information
528  if (dims_[cpos] != img.dims_[cpos]) {
529  if (dims_[cpos] == 1) {
530  colortype_ = ColorType::GRAY;
531  }
532  else {
533  channels_[cpos] = 'o';
534  colortype_ = ColorType::none;
535  }
536  }
537  }
538 
539  data_ = img.Ptr(start);
540  datasize_ = 0; // This is set to zero, because when the View is not contiguous, it's useless to relay on this information
541  contiguous_ = false;
542  }

549  void Create(const std::vector<int>& dims, std::string channels, ColorType colortype, uint8_t* ptr, const std::vector<float>& spacings = std::vector<float>())
550  {
551  // Compute datasize
552  size_t new_datasize = DataTypeSize(DT);
553  new_datasize = std::accumulate(begin(dims), end(dims), new_datasize, std::multiplies<size_t>());
554 
555  if (datasize_ != new_datasize) {
556  datasize_ = new_datasize;
557  }
558 
559  elemtype_ = DT;
560  elemsize_ = DataTypeSize(elemtype_);
561  dims_ = dims; // A check could be added to save this copy
562  spacings_ = spacings;
563  channels_ = std::move(channels);
564  colortype_ = colortype;
565  datasize_ = new_datasize;
566 
567  // Compute strides
568  strides_ = { elemsize_ };
569  int dsize = dims_.size();
570  for (int i = 0; i < dsize - 1; ++i) {
571  strides_.push_back(strides_[i] * dims_[i]);
572  }
573 
574  data_ = ptr;
575  mem_ = ShallowMemoryManager::GetInstance();
576  return;
577  }
