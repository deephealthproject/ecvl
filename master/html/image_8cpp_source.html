20 void Image::Create(const std::vector<int>& dims, DataType elemtype, std::string channels, ColorType colortype, const std::vector<float>& spacings)
21 {
22  if (IsEmpty() || !IsOwner()) {
23  *this = Image(dims, elemtype, std::move(channels), colortype, spacings);
24  return;
25  }
26  else {
27  if (!contiguous_) {
28  *this = Image(dims, elemtype, std::move(channels), colortype, spacings);
29  return;
30  }
31  else {
32  // Compute datasize
33  size_t new_datasize = DataTypeSize(elemtype);
34  new_datasize = std::accumulate(begin(dims), end(dims), new_datasize, std::multiplies<size_t>());
35 
36  if (datasize_ != new_datasize) {
37  datasize_ = new_datasize;
38  mem_->Deallocate(data_);
39  data_ = mem_->Allocate(new_datasize);
40  }
41 
42  elemtype_ = elemtype;
43  elemsize_ = DataTypeSize(elemtype_);
44  dims_ = dims; // A check could be added to save this copy
45  spacings_ = spacings;
46  channels_ = std::move(channels);
47  colortype_ = colortype;
48  datasize_ = new_datasize;
49 
50  // Compute strides
51  strides_ = { elemsize_ };
52  int dsize = dims_.size();
53  for (int i = 0; i < dsize - 1; ++i) {
54  strides_.push_back(strides_[i] * dims_[i]);
55  }
56  return;
57  }
58  }

61 void RearrangeAndCopy(const Image& src, Image& dst, const std::string& channels, DataType new_type)
62 {
63  if (src.elemtype_ == DataType::none)
64  throw std::runtime_error("Why should you copy a Image with none DataType into another?");
65 
66  if (&src == &dst) {
67  if (src.elemtype_ != new_type && new_type != DataType::none) {
68  throw std::runtime_error("src and dst cannot be the same image while changing the type");
69  return;
70  }
71  if (src.channels_ == channels) {
72  return;
73  }
74  }
75 
76  if (channels.size() != src.channels_.size()) {
77  ECVL_ERROR_WRONG_PARAMS("channels.size() does not match src.channels_.size()")
78  }
79 
80  if (src.channels_ == channels) {
81  CopyImage(src, dst, new_type);
82  return;
83  }
84 
85  if (src.elemtype_ == new_type) {
86  RearrangeChannels(src, dst, channels);
87  return;
88  }
89 
90  // bindings[new_pos] = old_pos
91  std::vector<int> bindings(src.channels_.size());
92  std::vector<int> new_dims(src.dims_);
93  std::vector<float> new_spacings(src.spacings_.size());
94 
95  // Check if rearranging is required
96  for (size_t old_pos = 0; old_pos < src.channels_.size(); old_pos++) {
97  char c = src.channels_[old_pos];
98  size_t new_pos = channels.find(c);
99  if (new_pos == std::string::npos) {
100  ECVL_ERROR_WRONG_PARAMS("channels contains wrong characters")
101  }
102  else {
103  bindings[new_pos] = old_pos;
104  new_dims[new_pos] = src.dims_[old_pos];
105  if (new_spacings.size() == new_dims.size()) { // spacings is not a mandatory field
106  new_spacings[new_pos] = src.spacings_[old_pos];
107  }
108  }
109  }
110 
111  if (new_type == DataType::none) {
112  // Get type from dst or src
113  if (dst.IsEmpty()) {
114  RearrangeChannels(src, dst, channels);
115  return;
116  }
117  if (src.dims_ != dst.dims_ || dst.channels_ != channels || src.elemtype_ != dst.elemtype_) {
118  // Destination needs to be resized
119  if (dst.mem_ == ShallowMemoryManager::GetInstance()) {
120  throw std::runtime_error("Trying to resize an Image which doesn't own data.");
121  }
122  if (src.dims_ != dst.dims_ || dst.channels_ != channels || src.elemsize_ != dst.elemsize_) {
123  dst = Image(new_dims, dst.elemtype_ == DataType::none ? src.elemtype_ : dst.elemtype_, channels, src.colortype_);
124  }
125  }
126  if (src.colortype_ != dst.colortype_) {
127  // Destination needs to change its color space
128  if (dst.mem_ == ShallowMemoryManager::GetInstance()) {
129  throw std::runtime_error("Trying to change color space on an Image which doesn't own data.");
130  }
131  dst.colortype_ = src.colortype_;
132  }
133  }
134  else {
135  if (dst.IsEmpty()) {
136  dst = Image(new_dims, new_type, channels, src.colortype_);
137  }
138  else {
139  if (src.dims_ != dst.dims_ || dst.channels_ != channels || dst.elemtype_ != new_type) {
140  // Destination needs to be resized
141  if (dst.mem_ == ShallowMemoryManager::GetInstance()) {
142  throw std::runtime_error("Trying to resize an Image which doesn't own data.");
143  }
144  if (src.dims_ != dst.dims_ || dst.channels_ != channels || dst.elemsize_ != DataTypeSize(new_type)) {
145  dst = Image(new_dims, new_type, channels, src.colortype_);
146  }
147  else {
148  dst.elemtype_ = new_type;
149  }
150  }
151  if (src.colortype_ != dst.colortype_) {
152  // Destination needs to change its color space
153  if (dst.mem_ == ShallowMemoryManager::GetInstance()) {
154  throw std::runtime_error("Trying to change color space on an Image which doesn't own data.");
155  }
156  dst.colortype_ = src.colortype_;
157  }
158  }
159  }
160 
161  static constexpr Table2D<StructRearrangeImage> table;
162  table(src.elemtype_, dst.elemtype_)(src, dst, bindings);
163 }
