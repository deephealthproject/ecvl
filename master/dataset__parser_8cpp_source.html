48 {

52  static std::regex r{ R"(https?://.*)" };
53 
54  for (int i = 0; i < location_.size(); ++i) {
55  auto location = is_gt ? label_path_.value() : location_[i];
56  if (std::regex_match(location.string(), r)) {
57  // TODO: Use libcurl instead of system call

60  if (system(cmd.c_str()) != 0) {

63  ECVL_ERROR_CANNOT_LOAD_FROM_URL

65  else {

68  }

72  ECVL_ERROR_FILE_DOES_NOT_EXIST

76

97 void Dataset::DecodeImages(const YAML::Node& node, const path& root_path, bool verify)
98 {
99  // Allocate memory for the images
100  this->samples_.resize(node.size());
101  int counter = -1;
102  // RegEx which matchs URLs
103  std::regex r{ R"(https?://.*)" };
104 
105  for (auto& n : node) {
106  // iterate over images
107  auto& sample = this->samples_[++counter];
108  sample.location_ = vector<path>();
109 
110  if (n.IsScalar()) {
111  // locations is provided as scalar without label or values
112  sample.location_.push_back(n.as<string>());
113  }
114  else if (n.IsSequence()) {
115  sample.location_.resize(n.size());
116  for (int i = 0; i < n.size(); ++i) {
117  sample.location_[i] = n[i].as<string>();
118  }
119  }
120  else { // location optionally has labels and values
121  if (n["location"].IsSequence()) {
122  // values is a list
123  sample.location_.resize(n["location"].size());
124  for (int i = 0; i < n["location"].size(); ++i) {
125  sample.location_[i] = n["location"][i].as<string>();
126  }
127  }
128  else {
129  sample.location_.push_back(n["location"].as<string>());
130  }
131 
132  // Load labels
133  if (n["label"].IsDefined()) {
134  sample.label_ = vector<int>();
135  if (n["label"].IsSequence()) {
136  // values is a list
137  for (int i = 0; i < n["label"].size(); ++i) {
138  FindLabel(sample, n["label"][i]);
139  }
140  }
141  else {
142  // label is a single value
143  FindLabel(sample, n["label"]);
144  }
145  }
146 
147  // Load features
148  if (n["values"].IsDefined()) {
149  sample.values_.emplace(map<int, string>());
150  if (n["values"].IsSequence()) {
151  // values is a list
152  for (int i = 0; i < n["values"].size(); ++i) {
153  if (!n["values"][i].IsNull()) {
154  sample.values_.value()[i] = n["values"][i].as<string>();
155  }
156  }
157  }
158  else {
159  // values is a dictionary
160  for (YAML::const_iterator it = n["values"].begin(); it != n["values"].end(); ++it) {
161  sample.values_.value()[this->features_map_[it->first.as<string>()]] = it->second.as<string>();
162  }
163  }
164  }
165  }
166 
167  for (int i = 0; i < sample.location_.size(); ++i) {
168  if (sample.location_[i].is_relative() && !std::regex_match(sample.location_[i].string(), r)) {
169  // Convert relative path to absolute
170  sample.location_[i] = root_path / sample.location_[i];
171  if (sample.label_path_.has_value()) {
172  sample.label_path_ = root_path / sample.label_path_.value();
173  }
174  }
175  if (verify) {
176  if (!filesystem::exists(sample.location_[i])) {
177  cerr << ECVL_WARNING_MSG "sample file " << sample.location_[i] << " does not exist" << endl;
178  }
179  if (sample.label_path_.has_value()) {
180  if (!filesystem::exists(sample.label_path_.value())) {
181  cerr << ECVL_WARNING_MSG "label file " << sample.label_path_.value() << " does not exist" << endl;
182  }
183  }
184  }
185  }
186  }
187 }

246 {

253 
254  YAML::Node config;
255  try {

260  cerr << "MSG: " << e.what();
261  ECVL_ERROR_NOT_REACHABLE_CODE
262  }
263 
264  if (config["name"].IsDefined()) {
265  this->name_ = config["name"].as<string>();
266  }
267  if (config["description"].IsDefined()) {
268  this->description_ = config["description"].as<string>();
269  }
270  if (config["classes"].IsDefined()) {
271  this->classes_ = config["classes"].as<vector<string>>();
272  }
273  if (config["features"].IsDefined()) {
274  this->features_ = config["features"].as<vector<string>>();
275  for (int i = 0; i < this->features_.size(); ++i) {
276  this->features_map_[this->features_[i]] = i;
277  }
278  }
279 
280  DecodeImages(config["images"], abs_filename.parent_path(), verify);
281  if (config["split"].IsDefined()) {
282  this->split_ = config["split"].as<Split>();
283  }
284 }
