174 // Template specialization for the in-place subtraction between scalar and Image.
175 template<DataType DT, typename T>
176 struct ScalarImageSubImpl {
177  static void _(T value, Image& img, bool saturate)
178  {
179  View<DT> v(img);
180  auto i = v.Begin(), e = v.End();
181  for (; i != e; ++i) {
182  auto& p = *i;
183  if (saturate) {
184  p = saturate_cast<DT>(PromoteSub(value, p));

186  else {
187  p = static_cast<typename TypeInfo<DT>::basetype>(value - p);

189  }

349 // Template specialization for the in-place division between scalar and Image.
350 template<DataType DT, typename T, typename ET>
351 struct ScalarImageDivImpl {
352  static void _(T value, Image& img, bool saturate, ET epsilon)
353  {

356  for (; i != e; ++i) {
357  auto& p = *i;
358  if (saturate) {
359  p = saturate_cast<DT>(PromoteDiv(value, (p + epsilon)));

361  else {
