81 void ImageToTensor(const Image& img, tensor& t)
82 {
83  Image tmp;
84

86  case 3:
87  if (img.channels_ != "xyc") {
88  RearrangeChannels(img, tmp, "xyc", DataType::float32);

97  RearrangeChannels(img, tmp, "xyzc", DataType::float32);
98  }
99  else {
100  CopyImage(img, tmp, DataType::float32);

102  t = eddlT::create({ tmp.dims_[2], tmp.dims_[3], tmp.dims_[1], tmp.dims_[0] });

111 void ImageToTensor(Image& img, tensor& t, const int& offset)
112 {
113  Image tmp;
114  int tot_dims = 0;
115 
116  switch (img.dims_.size()) {
117  case 3:
118  if (img.channels_ != "xyc")
119  RearrangeChannels(img, tmp, "xyc", DataType::float32);
120  tot_dims = img.dims_[0] * img.dims_[1] * img.dims_[2];
121  break;
122  case 4:
123  if (img.channels_ != "xyzc")
124  RearrangeChannels(img, tmp, "xyzc", DataType::float32);
125  tot_dims = img.dims_[0] * img.dims_[1] * img.dims_[2] * img.dims_[3];
126  break;
127  default:
128  ECVL_ERROR_MSG "Image must have 3 or 4 dimensions";
129  }
130 
131  if (tmp.elemtype_ != DataType::float32) {
132  CopyImage(img, tmp, DataType::float32);
133  }
134 
135  memcpy(t->ptr + tot_dims * offset, tmp.data_, tot_dims * sizeof(float));
136 }

140 void DatasetToTensor(const Dataset& dataset, const std::vector<int>& size, const std::vector<int>& split, tensor& images, tensor& labels, ColorType ctype)
141 {
142  if (size.size() != 2) {
143  ECVL_ERROR_MSG "size must have 2 dimensions (height, width)";
144  }
145  Image tmp;
146 
147  int n_samples = split.size();
148  int n_channels = dataset.samples_[0].LoadImage(ctype, false).Channels();
149  int n_classes = static_cast<int>(dataset.classes_.size());
150  // Allocate memory for EDDL tensors
151  images = eddlT::create({ n_samples, n_channels, size[0], size[1] });
152  labels = eddlT::create({ n_samples, n_classes });
153 
154  // Fill tensors with data
155  int i = 0;
156  for (auto& index : split) {
157  const Sample& elem = dataset.samples_[index];
158  // Copy image into tensor (images)
159  ResizeDim(elem.LoadImage(ctype, false), tmp, { size[1], size[0] });
160  ImageToTensor(tmp, images, i);
161  if (elem.label_) {
162  // Copy labels into tensor (labels)
163  vector<float> l(n_classes, 0);
164  for (int j = 0; j < elem.label_.value().size(); ++j) {
165  l[elem.label_.value()[j]] = 1;
166  }
167  memcpy(labels->ptr + l.size() * i, l.data(), l.size() * sizeof(float));
168  }
169  ++i;
170  }
171 }

225 void DLDataset::LoadBatch(tensor& images, tensor& labels)
226 {
227  if (resize_dims_.size() != 2) {
228  ECVL_ERROR_MSG "resize_dims_ must have 2 dimensions (height, width)";
229  }
230 
231  int& bs = batch_size_;
232  Image tmp;
233  int offset = 0, start = 0;
234 
235  // Move to next samples
236  start = current_batch_[current_split_] * bs;
237  current_batch_[current_split_]++;
238 
239  // Fill tensors with data
240  for (int i = start; i < start + bs; ++i) {
241  const int index = GetSplit()[i];
242  const Sample& elem = samples_[index];
243  // Read and resize (HxW -> WxH) image
244  ResizeDim(elem.LoadImage(ctype_, false), tmp, { resize_dims_[1], resize_dims_[0] });
245  // Copy image into tensor (images)
246  ImageToTensor(tmp, images, offset);
247 
248  if (elem.label_) {
249  // Copy labels into tensor (labels)
250  vector<float> lab(classes_.size(), 0);
251  for (int j = 0; j < elem.label_.value().size(); ++j) {
252  lab[elem.label_.value()[j]] = 1;
253  }
254  memcpy(labels->ptr + lab.size() * offset, lab.data(), lab.size() * sizeof(float));
255  }
256  else if (elem.label_path_) {
257  // Read and resize (HxW -> WxH) ground truth image
258  ResizeDim(elem.LoadImage(ctype_gt_, true), tmp, { resize_dims_[1], resize_dims_[0] });
259  // Copy labels into tensor (labels)
260  ImageToTensor(tmp, labels, offset);
261  }
262  ++offset;
263  }
