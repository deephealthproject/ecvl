113  void SetLastBatch(int batch_size)
114  {
115  // last batch is the remainder of the number of samples of the split divided by the batch size.
116  // if drop last is true or the remainder is 0, last batch is equal to the batch size.
117  auto value = vsize(samples_indices_) % batch_size;
118  last_batch_ = drop_last_ ? batch_size : (value == 0 ? batch_size : value);
119  }
