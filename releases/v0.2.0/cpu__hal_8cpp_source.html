53 struct StructRearrangeImage
54 {
55  static void _(const Image& src, Image& dst, const std::vector<int>& bindings)
56  {
57  using dsttype = typename TypeInfo<DDT>::basetype;
58  ConstView<SDT> vsrc(src);
59  View<DDT> vdst(dst);
60  auto id = vdst.Begin();
61 
62  for (size_t tmp_pos = 0; tmp_pos < dst.datasize_; tmp_pos += dst.elemsize_, ++id) {
63  int x = static_cast<int>(tmp_pos);
64  int src_pos = 0;
65  for (int i = vsize(dst.dims_) - 1; i >= 0; i--) {
66  src_pos += (x / dst.strides_[i]) * src.strides_[bindings[i]];
67  x %= dst.strides_[i];
68  }
69 
70  *id = static_cast<dsttype>(*(vsrc.data_ + src_pos));
71  }
72  }
