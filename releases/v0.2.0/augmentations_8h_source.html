32 class param {
33  static std::istream& read_until(std::istream& is, std::string& s, const std::string& list) {
34  s.clear();
35  while (is.peek() && is && list.find(is.peek()) == list.npos) {
36  s += is.get();
37  }
38  return is;
39  }

41  void read_vals(std::istream& is, char closing_char) {
42  double val;
43  char next_char;
44  do {
45  is >> val;
46  if (!is) {
47  break;
48  }
49  vals_.push_back(val);
50  is >> next_char;
51  } while (next_char == ',');
52  if (!is || next_char != closing_char) {
53  std::cerr << "Error while reading values of parameter " << name_ << "\n"; // TODO: standardize
54  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
55  }
56  }
57 public:
58  enum class type { range, vector, number, string };
59 
60  std::string name_;
61  type type_;
62  std::vector<double> vals_;
63  std::string str_;
64 
65  param() {}
66  param(std::istream& is) {
67  is >> std::ws;
68  read_until(is, name_, " =");
69  char next_char;
70  is >> std::ws >> next_char;
71  if (next_char != '=') {
72  throw std::runtime_error("Cannot read parameter name"); // TODO: standardize
73  }
74  is >> std::ws;
75  next_char = is.peek();
76  if (next_char == '[') { // range
77  is.ignore();
78  type_ = type::range;
79  read_vals(is, ']');
80  }
81  else if (next_char == '(') { // vector
82  is.ignore();
83  type_ = type::vector;
84  read_vals(is, ')');
85  }
86  else if (next_char == '"') { // string
87  is.ignore();
88  type_ = type::string;
89  std::getline(is, str_, '"');
90  }
91  else {
92  type_ = type::number;
93  vals_.resize(1);
94  is >> vals_[0];
95  }
96  if (!is) {
97  std::cerr << "Error while reading value of parameter " << name_ << "\n"; // TODO: standardize
98  throw std::runtime_error("Cannot read parameter value"); // TODO: standardize
99  }
100  }
101 
102  static auto read(std::istream& is) {
103  std::unordered_map<std::string, param> m;
104  is >> std::ws;
105  while (is.peek() != '\n') {
106  param p(is);
107  m[p.name_] = p;
108  }
109  return m;
110  }

214  break;

217  }
